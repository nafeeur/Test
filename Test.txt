from FlightRadar24 import FlightRadar24API

class CustomFlightRadar24API(FlightRadar24API):
    def __init__(self, api_key):
        super().__init__()
        self.api_key = api_key

    def fetch_with_key(self, url, params=None):
        headers = {"Authorization": f"Bearer {self.api_key}"}
        return self.session.get(url, params=params, headers=headers)

# Usage
api = CustomFlightRadar24API("your_api_key_here")
flights = api.fetch_with_key("https://enterprise.flightradar24.com/api/v1/flights", params={"bounds": "10.0,20.0,30.0,40.0"})
print(flights.json())
